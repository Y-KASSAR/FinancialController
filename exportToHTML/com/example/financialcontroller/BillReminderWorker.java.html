22   public class BillReminderWorker extends Worker {
23   
24       private static final String CHANNEL_ID = "bills_alerts";
25   
26       public BillReminderWorker(@NonNull Context context, @NonNull WorkerParameters workerParams) {
27           super(context, workerParams);
28       }
29   
30       @NonNull
31       @Override
32       public Result doWork() {
33           // 1. Get all reminders from DB
34           List<ReminderEntity> reminders = DatabaseClient.getInstance(getApplicationContext())
35                   .getAppDatabase()
36                   .reminderDao()
37                   .getAllReminders();
38   
39           // 2. Check each reminder
40           long today = System.currentTimeMillis();
41   
42           for (ReminderEntity reminder : reminders) {
43               if (reminder.isPaid) continue; // Skip paid bills
44   
45               long diffInMillis = reminder.dueDate - today;
46               long daysDiff = TimeUnit.MILLISECONDS.toDays(diffInMillis);
47   
48               // Logic: Check the checkboxes the user selected
49               boolean shouldNotify = false;
50               String message = "";
51   
52               // Case 1: Due Today (approx 0 days diff)
53               if (daysDiff == 0 && reminder.remindOnDate) {
54                   shouldNotify = true;
55                   message = "Due Today: " + reminder.title + " ($" + reminder.amount + ")";
56               }
57               // Case 2: Due Tomorrow (approx 1 day diff)
58               else if (daysDiff == 1 && reminder.remindDayBefore) {
59                   shouldNotify = true;
60                   message = "Due Tomorrow: " + reminder.title;
61               }
62               // Case 3: Due in a Week (approx 7 days diff)
63               else if (daysDiff == 7 && reminder.remindWeekBefore) {
64                   shouldNotify = true;
65                   message = "Due in 1 Week: " + reminder.title;
66               }
67   
68               if (shouldNotify) {
69                   sendNotification(reminder.id, "Bill Reminder", message);
70               }
71           }
72   
73           return Result.success();
74       }
75   
76       private void sendNotification(int id, String title, String message) {
77           NotificationManager manager = (NotificationManager) getApplicationContext().getSystemService(Context.NOTIFICATION_SERVICE);
78   
79           // Create Channel (Required for Android 8+)
80           if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
81               NotificationChannel channel = new NotificationChannel(CHANNEL_ID, "Bill Alerts", NotificationManager.IMPORTANCE_HIGH);
82               manager.createNotificationChannel(channel);
83           }
84   
85           // Intent to open App when clicked
86           Intent intent = new Intent(getApplicationContext(), Reminders.class);
87           PendingIntent pendingIntent = PendingIntent.getActivity(getApplicationContext(), 0, intent, PendingIntent.FLAG_IMMUTABLE);
88   
89           NotificationCompat.Builder builder = new NotificationCompat.Builder(getApplicationContext(), CHANNEL_ID)
90                   .setSmallIcon(R.drawable.ic_calendar) // Ensure this icon exists
91                   .setContentTitle(title)
92                   .setContentText(message)
93                   .setPriority(NotificationCompat.PRIORITY_HIGH)
94                   .setContentIntent(pendingIntent)
95                   .setAutoCancel(true);
96   
97           manager.notify(id, builder.build());
98       }
