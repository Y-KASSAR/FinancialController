1    package com.example.financialcontroller;
2    
3    import android.app.AlertDialog;
4    import android.graphics.Color;
5    import android.graphics.drawable.ColorDrawable;
6    import android.os.Bundle;
7    import android.os.Handler;
8    import android.os.Looper;
9    import android.view.LayoutInflater;
10   import android.view.View;
11   import android.widget.Button;
12   import android.widget.EditText;
13   import android.widget.Toast;
14   
15   import androidx.appcompat.app.AppCompatActivity;
16   import androidx.recyclerview.widget.LinearLayoutManager;
17   import androidx.recyclerview.widget.RecyclerView;
18   
19   import com.example.financialcontroller.data.CategoryEntity;
20   import com.example.financialcontroller.data.DatabaseClient;
21   import com.google.android.material.floatingactionbutton.FloatingActionButton;
22   
23   import java.util.List;
24   import java.util.concurrent.Executors;
25   
26   public class Categories extends AppCompatActivity {
27   
28       private RecyclerView recyclerView;
29       private FloatingActionButton fabAdd;
30       private CategoryAdapter adapter;
31   
32       @Override
33       protected void onCreate(Bundle savedInstanceState) {
34           super.onCreate(savedInstanceState);
35           setContentView(R.layout.activity_categories);
36   
37           recyclerView = findViewById(R.id.rv_categories);
38           recyclerView.setLayoutManager(new LinearLayoutManager(this));
39   
40           fabAdd = findViewById(R.id.fab_add_category);
41           fabAdd.setOnClickListener(v -> showAddCategoryDialog());
42   
43           loadCategories();
44       }
45   
46       private void loadCategories() {
47           Executors.newSingleThreadExecutor().execute(() -> {
48               List<CategoryEntity> categories = DatabaseClient.getInstance(getApplicationContext())
49                       .getAppDatabase()
50                       .categoryDao()
51                       .getAllCategories();
52   
53               new Handler(Looper.getMainLooper()).post(() -> {
54                   adapter = new CategoryAdapter(categories);
55                   recyclerView.setAdapter(adapter);
56               });
57           });
58       }
59   
60       private void showAddCategoryDialog() {
61           AlertDialog.Builder builder = new AlertDialog.Builder(this);
62           View view = LayoutInflater.from(this).inflate(R.layout.dialog_add_category, null);
63           builder.setView(view);
64   
65           AlertDialog dialog = builder.create();
66   
67           // Make background transparent so your CardView corners show nicely
68           if (dialog.getWindow() != null) {
69               dialog.getWindow().setBackgroundDrawable(new ColorDrawable(Color.TRANSPARENT));
70           }
71   
72           // Initialize Views from your new dialog XML
73           EditText etName = view.findViewById(R.id.et_category_name);
74           Button btnSave = view.findViewById(R.id.btn_save_cat);
75           Button btnCancel = view.findViewById(R.id.btn_cancel_cat);
76   
77           btnSave.setOnClickListener(v -> {
78               String name = etName.getText().toString().trim();
79               if (!name.isEmpty()) {
80                   saveCategory(name);
81                   dialog.dismiss();
82               } else {
83                   Toast.makeText(this, "Please enter a name", Toast.LENGTH_SHORT).show();
84               }
85           });
86   
87           btnCancel.setOnClickListener(v -> dialog.dismiss());
88   
89           dialog.show();
90       }
91   
92       private void saveCategory(String name) {
93           Executors.newSingleThreadExecutor().execute(() -> {
94               CategoryEntity newCategory = new CategoryEntity(name, "#4169E1", "default");
95               DatabaseClient.getInstance(getApplicationContext())
96                       .getAppDatabase()
97                       .categoryDao()
98                       .insert(newCategory);
99   
100              loadCategories(); // Refresh list
101          });
102      }
103  }
