1    package com.example.financialcontroller;
2    
3    import android.app.DatePickerDialog;
4    import android.os.Bundle;
5    import android.widget.Button;
6    import android.widget.CheckBox;
7    import android.widget.EditText;
8    import android.widget.Toast;
9    import androidx.appcompat.app.AppCompatActivity;
10   
11   import com.example.financialcontroller.data.DatabaseClient;
12   import com.example.financialcontroller.data.ReminderEntity;
13   
14   import java.text.SimpleDateFormat;
15   import java.util.Calendar;
16   import java.util.Locale;
17   
18   public class AddReminder extends AppCompatActivity {
19   
20       // Views
21       private EditText etTitle, etAmount, etDate;
22       private CheckBox cbWeekBefore, cbDayBefore, cbOnDueDate;
23       private Button btnSave;
24   
25       // Calendar for Date Picker
26       private final Calendar myCalendar = Calendar.getInstance();
27   
28       @Override
29       protected void onCreate(Bundle savedInstanceState) {
30           super.onCreate(savedInstanceState);
31           setContentView(R.layout.activity_add_reminder);
32   
33           // 1. Initialize Views (Matching your XML IDs)
34           etTitle = findViewById(R.id.et_bill_title);
35           etAmount = findViewById(R.id.et_bill_amount);
36           etDate = findViewById(R.id.et_bill_date);
37   
38           cbWeekBefore = findViewById(R.id.cb_week_before);
39           cbDayBefore = findViewById(R.id.cb_day_before);
40           cbOnDueDate = findViewById(R.id.cb_on_due_date);
41   
42           btnSave = findViewById(R.id.btn_save_reminder);
43   
44           // 2. Setup Date Picker Logic
45           DatePickerDialog.OnDateSetListener date = (view, year, month, day) -> {
46               myCalendar.set(Calendar.YEAR, year);
47               myCalendar.set(Calendar.MONTH, month);
48               myCalendar.set(Calendar.DAY_OF_MONTH, day);
49               updateDateLabel();
50           };
51   
52           etDate.setOnClickListener(v -> {
53               new DatePickerDialog(AddReminder.this, date,
54                       myCalendar.get(Calendar.YEAR),
55                       myCalendar.get(Calendar.MONTH),
56                       myCalendar.get(Calendar.DAY_OF_MONTH)).show();
57           });
58   
59           // 3. Handle Save Button
60           btnSave.setOnClickListener(v -> saveReminder());
61       }
62   
63       private void updateDateLabel() {
64           String myFormat = "MM/dd/yyyy"; // Format: 12/31/2023
65           SimpleDateFormat sdf = new SimpleDateFormat(myFormat, Locale.US);
66           etDate.setText(sdf.format(myCalendar.getTime()));
67       }
68   
69       private void saveReminder() {
70           // Get Inputs
71           String title = etTitle.getText().toString().trim();
72           String amountStr = etAmount.getText().toString().trim();
73           String dateStr = etDate.getText().toString().trim();
74   
75           // Validation
76           if (title.isEmpty() || amountStr.isEmpty() || dateStr.isEmpty()) {
77               Toast.makeText(this, "Please fill all fields", Toast.LENGTH_SHORT).show();
78               return;
79           }
80   
81           double amount = Double.parseDouble(amountStr);
82           long dueDate = myCalendar.getTimeInMillis();
83   
84           // Get Checkbox values
85           boolean weekBefore = cbWeekBefore.isChecked();
86           boolean dayBefore = cbDayBefore.isChecked();
87           boolean onDate = cbOnDueDate.isChecked();
88   
89           // Create Entity with ALL 7 arguments
90           ReminderEntity reminder = new ReminderEntity(
91                   title,
92                   amount,
93                   dueDate,
94                   false,       // isPaid (default false)
95                   weekBefore,  // remindWeekBefore
96                   dayBefore,   // remindDayBefore
97                   onDate       // remindOnDate
98           );
99   
100          // Save to Database
101          DatabaseClient.getInstance(getApplicationContext())
102                  .getAppDatabase()
103                  .reminderDao()
104                  .insert(reminder);
105  
106          // TODO: We will add the Notification Scheduler here later!
107  
108          Toast.makeText(this, "Reminder Saved!", Toast.LENGTH_SHORT).show();
109          finish(); // Close screen
110      }
111  }
