1    package com.example.financialcontroller;
2    
3    import android.view.LayoutInflater;
4    import android.view.View;
5    import android.view.ViewGroup;
6    import android.widget.ProgressBar;
7    import android.widget.TextView;
8    import androidx.annotation.NonNull;
9    import androidx.recyclerview.widget.RecyclerView;
10   import com.example.financialcontroller.data.BudgetEntity;
11   import java.util.List;
12   
13   public class BudgetAdapter extends RecyclerView.Adapter<BudgetAdapter.BudgetViewHolder> {
14   
15       // Helper class to hold data
16       public static class BudgetViewItem {
17           public BudgetEntity budget;
18           public double spent;
19   
20           public BudgetViewItem(BudgetEntity budget, double spent) {
21               this.budget = budget;
22               this.spent = spent;
23           }
24       }
25   
26       private List<BudgetViewItem> budgetList;
27   
28       public BudgetAdapter(List<BudgetViewItem> budgetList) {
29           this.budgetList = budgetList;
30       }
31   
32       @NonNull
33       @Override
34       public BudgetViewHolder onCreateViewHolder(@NonNull ViewGroup parent, int viewType) {
35           View view = LayoutInflater.from(parent.getContext()).inflate(R.layout.item_budget, parent, false);
36           return new BudgetViewHolder(view);
37       }
38   
39       @Override
40       public void onBindViewHolder(@NonNull BudgetViewHolder holder, int position) {
41           BudgetViewItem item = budgetList.get(position);
42   
43           holder.tvCategory.setText(item.budget.category);
44   
45           // Calculate progress
46           int progress = 0;
47           if (item.budget.limitAmount > 0) {
48               progress = (int) ((item.spent / item.budget.limitAmount) * 100);
49           }
50           holder.progressBar.setProgress(progress);
51           holder.tvPercent.setText(progress + "%");
52   
53           // Format text: "$425 spent of $500"
54           String status = String.format("$%.0f spent of $%.0f", item.spent, item.budget.limitAmount);
55           holder.tvAmount.setText(status);
56   
57           // Change color if over budget
58           if (item.spent > item.budget.limitAmount) {
59               holder.tvPercent.setTextColor(0xFFFF0000); // Red
60           } else {
61               holder.tvPercent.setTextColor(0xFFFF9800); // Orange
62           }
63       }
64   
65       @Override
66       public int getItemCount() {
67           return budgetList != null ? budgetList.size() : 0;
68       }
69   
70       static class BudgetViewHolder extends RecyclerView.ViewHolder {
71           TextView tvCategory, tvPercent, tvAmount;
72           ProgressBar progressBar;
73   
74           public BudgetViewHolder(@NonNull View itemView) {
75               super(itemView);
76               tvCategory = itemView.findViewById(R.id.tv_budget_category);
77               tvPercent = itemView.findViewById(R.id.tv_budget_percent);
78               tvAmount = itemView.findViewById(R.id.tv_budget_amount);
79               progressBar = itemView.findViewById(R.id.pb_budget);
80           }
81       }
82   }
