1    package com.example.financialcontroller;
2    
3    import android.content.Intent;
4    import android.os.Bundle;
5    import android.widget.Button;
6    import android.widget.EditText;
7    import android.widget.Toast;
8    import androidx.work.PeriodicWorkRequest;
9    import androidx.work.WorkManager;
10   import java.util.concurrent.TimeUnit;
11   import androidx.appcompat.app.AppCompatActivity;
12   
13   public class MainActivity extends AppCompatActivity {
14   
15       private EditText etUsername, etPassword;
16       private Button btnLogin;
17   
18       @Override
19       protected void onCreate(Bundle savedInstanceState) {
20           super.onCreate(savedInstanceState);
21           setContentView(R.layout.activity_main);
22   
23           // 1. Initialize Views
24           // Note: Ensure these IDs match your activity_main.xml
25           etUsername = findViewById(R.id.et_username);
26           etPassword = findViewById(R.id.et_password);
27           btnLogin = findViewById(R.id.btn_login);
28   
29           // 2. Handle Login Click
30           btnLogin.setOnClickListener(v -> {
31               String username = etUsername.getText().toString().trim();
32               String password = etPassword.getText().toString().trim();
33   
34               if (validateLogin(username, password)) {
35                   // Login Success
36                   Toast.makeText(MainActivity.this, "Welcome back!", Toast.LENGTH_SHORT).show();
37   
38                   // Navigate to Dashboard
39                   Intent intent = new Intent(MainActivity.this, DashboardActivity.class);
40                   startActivity(intent);
41                   finish(); // Prevent going back to login screen
42               }
43           });
44           // Schedule the Bill Reminder Worker to run once every 24 hours
45           PeriodicWorkRequest reminderRequest = new PeriodicWorkRequest.Builder(BillReminderWorker.class, 24, TimeUnit.HOURS)
46                   .build();
47   
48           WorkManager.getInstance(this).enqueue(reminderRequest);
49       }
50   
51       private boolean validateLogin(String username, String password) {
52           if (username.isEmpty()) {
53               etUsername.setError("Username is required");
54               return false;
55           }
56           if (password.isEmpty()) {
57               etPassword.setError("Password is required");
58               return false;
59           }
60           // For a final project, you can add a hardcoded check like:
61           // if (username.equals("admin") && password.equals("1234")) return true;
62   
63           return true; // Allow any input for now
64       }
65   }
