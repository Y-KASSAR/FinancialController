20   public class TransactionAdapter extends RecyclerView.Adapter<TransactionAdapter.TransactionViewHolder> {
21   
22       private Context context;
23       private List<TransactionEntity> transactionList;
24   
25       public TransactionAdapter(Context context, List<TransactionEntity> transactionList) {
26           this.context = context;
27           this.transactionList = transactionList;
28       }
29   
30       @NonNull
31       @Override
32       public TransactionViewHolder onCreateViewHolder(@NonNull ViewGroup parent, int viewType) {
33           View view = LayoutInflater.from(context).inflate(R.layout.item_transaction, parent, false);
34           return new TransactionViewHolder(view);
35       }
36   
37       @Override
38       public void onBindViewHolder(@NonNull TransactionViewHolder holder, int position) {
39           TransactionEntity transaction = transactionList.get(position);
40   
41           // 1. Set Category & Note
42           holder.tvCategory.setText(transaction.category);
43   
44           // 2. Format Date (e.g., "Today, 10:30 AM")
45           SimpleDateFormat sdf = new SimpleDateFormat("MMM dd, hh:mm a", Locale.getDefault());
46           holder.tvDate.setText(sdf.format(new Date(transaction.date)));
47   
48           // 3. Set Amount & Color
49           if (transaction.type.equals("INCOME")) {
50               holder.tvAmount.setText("+ $" + String.format("%.2f", transaction.amount));
51               holder.tvAmount.setTextColor(ContextCompat.getColor(context, R.color.income_green));
52           } else {
53               holder.tvAmount.setText("- $" + String.format("%.2f", transaction.amount));
54               holder.tvAmount.setTextColor(ContextCompat.getColor(context, R.color.expense_red));
55           }
56   
57           // 4. Set Icon (Optional: You can add logic here to change icon based on category)
58           // holder.ivIcon.setImageResource(getIconForCategory(transaction.category));
59       }
60   
61       @Override
62       public int getItemCount() {
63           return transactionList.size();
64       }
65   
66       public class TransactionViewHolder extends RecyclerView.ViewHolder {
67           TextView tvCategory, tvDate, tvAmount;
68           ImageView ivIcon;
69   
70           public TransactionViewHolder(@NonNull View itemView) {
71               super(itemView);
72               tvCategory = itemView.findViewById(R.id.tv_category);
73               tvDate = itemView.findViewById(R.id.tv_date);
74               tvAmount = itemView.findViewById(R.id.tv_amount);
75               ivIcon = itemView.findViewById(R.id.iv_category_icon);
76           }
77       }
