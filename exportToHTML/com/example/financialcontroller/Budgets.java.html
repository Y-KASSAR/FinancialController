57       private void setupBottomNavigation() {
58           BottomNavigationView bottomNavigationView = findViewById(R.id.bottom_navigation);
59           bottomNavigationView.setSelectedItemId(R.id.nav_budget); // Highlight "Budgets"
60   
61           bottomNavigationView.setOnItemSelectedListener(item -> {
62               int id = item.getItemId();
63               if (id == R.id.nav_home) {
64                   startActivity(new Intent(getApplicationContext(), DashboardActivity.class));
65                   overridePendingTransition(0, 0);
66                   return true;
67               } else if (id == R.id.nav_analysis) {
68                   startActivity(new Intent(getApplicationContext(), AnalysisActivity.class));
69                   overridePendingTransition(0, 0);
70                   return true;
71               } else if (id == R.id.nav_budget) {
72                   return true; // Already here
73               }
74               return false;
75           });
76       }

78       private void loadBudgets() {
79           Executors.newSingleThreadExecutor().execute(() -> {
80               // 1. Get all budgets
81               List<BudgetEntity> budgets = DatabaseClient.getInstance(getApplicationContext())
82                       .getAppDatabase()
83                       .budgetDao()
84                       .getAllBudgets();
85   
86               List<BudgetAdapter.BudgetViewItem> viewItems = new ArrayList<>();
87   
88               for (BudgetEntity b : budgets) {
89                   // FIX: Use Double wrapper to catch NULLs
90                   Double spentObj = DatabaseClient.getInstance(getApplicationContext())
91                           .getAppDatabase()
92                           .transactionDao()
93                           .getSpentByCategory(b.category);
94   
95                   // If null (no expenses yet), default to 0.0
96                   double spent = (spentObj == null) ? 0.0 : spentObj;
97   
98                   viewItems.add(new BudgetAdapter.BudgetViewItem(b, spent));
99               }
100  
101              // 2. Update UI
102              new Handler(Looper.getMainLooper()).post(() -> {
103                  adapter = new BudgetAdapter(viewItems);
104                  recyclerView.setAdapter(adapter);
105              });
106          });
107      }

109      private void showAddBudgetDialog() {
110          AlertDialog.Builder builder = new AlertDialog.Builder(this);
111          View v = getLayoutInflater().inflate(R.layout.dialog_add_budget, null);
112  
113          // 1. Init views
114          Spinner spinner = v.findViewById(R.id.spinner_budget_category);
115          EditText etLimit = v.findViewById(R.id.et_budget_limit);
116          Button btnSave = v.findViewById(R.id.btn_save_budget);
117          Button btnCancel = v.findViewById(R.id.btn_cancel_budget);
118  
119          // 2. Setup Spinner
120          ArrayAdapter<CharSequence> adapter = ArrayAdapter.createFromResource(this,
121                  R.array.categories_array, android.R.layout.simple_spinner_item);
122          adapter.setDropDownViewResource(android.R.layout.simple_spinner_dropdown_item);
123          spinner.setAdapter(adapter);
124  
125          // 3. Create Dialog
126          AlertDialog dialog = builder.setView(v).create();
127  
128          // 4. Button Logic
129          btnSave.setOnClickListener(view -> {
130              String category = spinner.getSelectedItem().toString();
131              String limitStr = etLimit.getText().toString();
132  
133              if (!limitStr.isEmpty()) {
134                  saveBudget(category, Double.parseDouble(limitStr));
135                  dialog.dismiss();
136              } else {
137                  etLimit.setError("Required");
138              }
139          });
140  
141          btnCancel.setOnClickListener(view -> dialog.dismiss());
142  
143          dialog.show();
144      }
145      private void saveBudget(String category, double limit) {
146          Executors.newSingleThreadExecutor().execute(() -> {
147              // Check if budget exists
148              BudgetEntity existing = DatabaseClient.getInstance(getApplicationContext())
149                      .getAppDatabase()
150                      .budgetDao()
151                      .getBudgetByCategory(category);
152  
153              if (existing != null) {
154                  existing.limitAmount = limit;
155                  DatabaseClient.getInstance(getApplicationContext())
156                          .getAppDatabase()
157                          .budgetDao()
158                          .update(existing);
159              } else {
160                  BudgetEntity newBudget = new BudgetEntity(category, limit);
161                  DatabaseClient.getInstance(getApplicationContext())
162                          .getAppDatabase()
163                          .budgetDao()
164                          .insert(newBudget);
165              }
166  
167              // Refresh List
168              loadBudgets();
169          });
170      }
