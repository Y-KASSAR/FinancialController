1    package com.example.financialcontroller;
2    
3    import android.os.Bundle;
4    import android.view.View;
5    import android.widget.TextView;
6    import androidx.appcompat.app.AppCompatActivity;
7    import androidx.recyclerview.widget.LinearLayoutManager;
8    import androidx.recyclerview.widget.RecyclerView;
9    
10   import com.example.financialcontroller.data.DatabaseClient;
11   import com.example.financialcontroller.data.TransactionEntity;
12   
13   import java.util.List;
14   
15   public class SearchResultsActivity extends AppCompatActivity {
16   
17       private RecyclerView recyclerView;
18       private TextView tvCount, tvNoResults;
19       private TransactionAdapter adapter;
20   
21       @Override
22       protected void onCreate(Bundle savedInstanceState) {
23           super.onCreate(savedInstanceState);
24           setContentView(R.layout.activity_search_results);
25   
26           recyclerView = findViewById(R.id.rv_search_results);
27           tvCount = findViewById(R.id.tv_result_count);
28           tvNoResults = findViewById(R.id.tv_no_results);
29   
30           recyclerView.setLayoutManager(new LinearLayoutManager(this));
31   
32           // Get data passed from SearchFilter
33           String keyword = getIntent().getStringExtra("KEYWORD");
34           String type = getIntent().getStringExtra("TYPE");
35           String category = getIntent().getStringExtra("CATEGORY");
36           long fromDate = getIntent().getLongExtra("FROM", 0);
37           long toDate = getIntent().getLongExtra("TO", 0);
38   
39           performSearch(keyword, type, category, fromDate, toDate);
40       }
41   
42       private void performSearch(String keyword, String type, String category, long from, long to) {
43           new Thread(() -> {
44               List<TransactionEntity> results = DatabaseClient.getInstance(getApplicationContext())
45                       .getAppDatabase()
46                       .transactionDao()
47                       .searchTransactions(type, category, from, to, keyword);
48   
49               runOnUiThread(() -> {
50                   if (results.isEmpty()) {
51                       tvNoResults.setVisibility(View.VISIBLE);
52                       recyclerView.setVisibility(View.GONE);
53                       tvCount.setText("Found 0 transactions");
54                   } else {
55                       tvNoResults.setVisibility(View.GONE);
56                       recyclerView.setVisibility(View.VISIBLE);
57                       tvCount.setText("Found " + results.size() + " transactions");
58   
59                       // FIX: Context first, then List
60                       adapter = new TransactionAdapter(SearchResultsActivity.this, results);
61                       recyclerView.setAdapter(adapter);
62                   }
63               });
64           }).start();
65       }
66   }
